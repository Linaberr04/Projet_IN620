// Machine de Turing a 1 ruban compatible avec la MTU de la question 9.
// Alphabet utilise : 0, 1, _ uniquement.
// Idee : on efface le premier symbole, puis on va verifier le dernier symbole restant.
// Si les deux correspondent, on efface aussi le dernier symbole et on recommence.
// Si ca ne correspond pas, la machine se bloque.

// Chercher le premier symbole non efface a gauche
q0,0
q_right_0,_,>

q0,1
q_right_1,_,>

q0,_
accept,_,-

// On a lu 0 a gauche : aller au blanc de droite
q_right_0,0
q_right_0,0,>

q_right_0,1
q_right_0,1,>

q_right_0,_
q_check_0,_,<

// Le dernier symbole restant doit etre 0
q_check_0,0
q_back,_,<

q_check_0,_
accept,_,-

// Pas de transition q_check_0,1 : ce n'est pas un palindrome

// On a lu 1 a gauche : aller au blanc de droite
q_right_1,0
q_right_1,0,>

q_right_1,1
q_right_1,1,>

q_right_1,_
q_check_1,_,<

// Le dernier symbole restant doit etre 1
q_check_1,1
q_back,_,<

q_check_1,_
accept,_,-

// Pas de transition q_check_1,0 : ce n'est pas un palindrome

// Revenir au blanc de gauche, puis repartir sur le prochain symbole non efface
q_back,0
q_back,0,<

q_back,1
q_back,1,<

q_back,_
q0,_,>
